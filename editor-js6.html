// ════════════════════════════════════════════════════════ // 設定タブ（editor-js6.html） // ════════════════════════════════════════════════════════ function initSettingsTab() { document.getElementById('save-settings-btn').addEventListener('click', saveSettings); document.getElementById('auto-setup-btn').addEventListener('click', runAutoSetup); // loadSettingsValues は switchTab('settings') 時に遅延ロードされる } function loadSettingsValues() { // 全画面オーバーレイではなくインライン表示（他タブへの切り替えをブロックしない） const container = document.querySelector('.settings-container'); let loadingMsg = document.getElementById('settingsLoadingMsg'); if (!loadingMsg && container) { loadingMsg = document.createElement('div'); loadingMsg.id = 'settingsLoadingMsg'; loadingMsg.textContent = '⏳ 設定を読み込み中...'; loadingMsg.style.cssText = 'padding: 10px; color: #666; font-size: 13px;'; container.prepend(loadingMsg); } google.script.run .withSuccessHandler(data => { if (loadingMsg) loadingMsg.remove(); if (!data.success) { alert('設定の読み込みに失敗しました: ' + (data.error || '不明なエラー')); return; } ['ENGLISHWORDS_FOLDER_ID', 'ENGLISHWORDS_SHEET_ID', 'GITHUB_BASE_URL', 'GITHUB_TOKEN', 'GOOGLE_CLOUD_TTS_API_KEY', 'HOMEPAGE_URL'].forEach(key => { const el = document.getElementById('setting-' + key); if (el) el.value = data[key] || ''; }); }) .withFailureHandler(err => { if (loadingMsg) loadingMsg.remove(); alert('設定の読み込みに失敗しました: ' + err); }) .getScriptPropertiesForSettings(); } function saveSettings() { const settings = {}; ['ENGLISHWORDS_FOLDER_ID', 'ENGLISHWORDS_SHEET_ID', 'GITHUB_BASE_URL', 'GITHUB_TOKEN', 'GOOGLE_CLOUD_TTS_API_KEY', 'HOMEPAGE_URL'].forEach(key => { const el = document.getElementById('setting-' + key); if (el) settings[key] = el.value.trim(); }); showLoadingIndicator('設定を保存中...'); google.script.run .withSuccessHandler(result => { hideLoadingIndicator(); if (result.success) { alert('✅ 設定を保存しました。'); } else { alert('❌ 保存に失敗しました: ' + (result.error || '不明なエラー')); } }) .withFailureHandler(err => { hideLoadingIndicator(); alert('❌ 保存に失敗しました: ' + err); }) .saveScriptProperties(settings); } function runAutoSetup() { if (!confirm('Drive フォルダとマスタースプレッドシートを自動作成しますか？\n既存データは変更されません（べき等）。')) return; showLoadingIndicator('自動セットアップ中...'); google.script.run .withSuccessHandler(result => { hideLoadingIndicator(); let msg = result.success ? '✅ 自動セットアップ完了\n\n' : '❌ エラー: ' + (result.error || '不明なエラー') + '\n\n'; if (result.created && result.created.length > 0) msg += '【新規作成】\n' + result.created.join('\n') + '\n\n'; if (result.existing && result.existing.length > 0) msg += '【既存（スキップ）】\n' + result.existing.join('\n') + '\n\n'; if (result.manualRequired && result.manualRequired.length > 0) msg += '⚠️ 手動設定が必要なプロパティ:\n' + result.manualRequired.join(', '); alert(msg); if (result.success) loadSettingsValues(); }) .withFailureHandler(err => { hideLoadingIndicator(); alert('❌ 自動セットアップに失敗しました: ' + err); }) .runSetupScriptProperties(); } function togglePasswordVisibility(inputId) { const input = document.getElementById(inputId); if (!input) return; input.type = (input.type === 'password') ? 'text' : 'password'; } function generateQrCode() { const url = document.getElementById('setting-HOMEPAGE_URL').value.trim(); if (!url) { alert('HOMEPAGE_URL を入力してください。'); return; } // Google Charts QR Code API（無料・API キー不要） const qrUrl = 'https://chart.googleapis.com/chart?cht=qr&chs=400x400&chl=' + encodeURIComponent(url) + '&choe=UTF-8'; const img = document.getElementById('qr-code-img'); const link = document.getElementById('qr-download-link'); if (img) img.src = qrUrl; if (link) link.href = qrUrl; const area = document.getElementById('qr-code-area'); if (area) area.style.display = 'block'; } // onclick 属性から呼べるようにグローバルに公開 window.togglePasswordVisibility = togglePasswordVisibility; window.generateQrCode = generateQrCode;
